SET SQL_SAFE_UPDATES = 0;
SET FOREIGN_KEY_CHECKS = 0;

-- Veritabanını sıfırla ve yeniden oluştur
DROP DATABASE IF EXISTS sanat_galerisi;
CREATE DATABASE sanat_galerisi CHARACTER SET utf8mb4 COLLATE utf8mb4_turkish_ci;
USE sanat_galerisi;

-- ═══════════════════════════════════════════════════════════════════
-- TABLOLAR
-- ═══════════════════════════════════════════════════════════════════

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    full_name VARCHAR(120) NOT NULL,
    email VARCHAR(180) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    role ENUM('kullanici','admin') DEFAULT 'kullanici',
    created_at DATETIME DEFAULT NOW()
);

CREATE TABLE artists (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(120) NOT NULL,
    bio TEXT,
    birth_year INT,
    nationality VARCHAR(80)
);

CREATE TABLE categories (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(80) NOT NULL
);

CREATE TABLE artworks (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    artist_id INT,
    category_id INT,
    description TEXT,
    price DECIMAL(12,2) NOT NULL,
    image_url VARCHAR(300),
    year_created INT,
    medium VARCHAR(100),
    dimensions VARCHAR(80),
    discount_percent INT DEFAULT 0,
    stock INT DEFAULT 1,
    created_at DATETIME DEFAULT NOW(),
    FOREIGN KEY (artist_id) REFERENCES artists(id),
    FOREIGN KEY (category_id) REFERENCES categories(id)
);

CREATE TABLE events (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    description TEXT,
    event_type ENUM('atolye','sergi','konferans','tur') DEFAULT 'atolye',
    event_date DATETIME NOT NULL,
    duration_minutes INT DEFAULT 120,
    location VARCHAR(200),
    capacity INT NOT NULL,
    price DECIMAL(10,2) DEFAULT 0,
    image_url VARCHAR(300),
    discount_percent INT DEFAULT 0,
    created_at DATETIME DEFAULT NOW()
);

CREATE TABLE favorites (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    artwork_id INT NOT NULL,
    created_at DATETIME DEFAULT NOW(),
    UNIQUE (user_id, artwork_id),
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (artwork_id) REFERENCES artworks(id)
);

CREATE TABLE reservations (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    event_id INT NOT NULL,
    participant_count INT DEFAULT 1,
    reservation_date DATE NOT NULL,
    status ENUM('beklemede','onaylandı','iptal') DEFAULT 'beklemede',
    created_at DATETIME DEFAULT NOW(),
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (event_id) REFERENCES events(id)
);

CREATE TABLE orders (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    artwork_id INT,
    event_id INT,
    amount DECIMAL(12,2) NOT NULL,
    payment_method ENUM('kredi_karti','banka_havalesi','kapida_odeme') DEFAULT 'kredi_karti',
    status ENUM('beklemede','onaylandı','kargoda','teslim_edildi','iptal') DEFAULT 'beklemede',
    created_at DATETIME DEFAULT NOW(),
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (artwork_id) REFERENCES artworks(id),
    FOREIGN KEY (event_id) REFERENCES events(id)
);

CREATE TABLE coupons (
    id INT AUTO_INCREMENT PRIMARY KEY,
    code VARCHAR(50) UNIQUE NOT NULL,
    discount_percent INT NOT NULL,
    is_active TINYINT DEFAULT 1,
    is_public TINYINT DEFAULT 1,
    expires_at DATE,
    description VARCHAR(200)
);

CREATE TABLE support_tickets (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    name VARCHAR(120) NOT NULL,
    email VARCHAR(180) NOT NULL,
    message TEXT NOT NULL,
    status ENUM('açık','işlemde','kapatıldı') DEFAULT 'açık',
    admin_reply TEXT DEFAULT NULL,
    replied_at DATETIME DEFAULT NULL,
    created_at DATETIME DEFAULT NOW(),
    FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE reviews (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    artwork_id INT,
    event_id INT,
    rating TINYINT NOT NULL CHECK (rating BETWEEN 1 AND 5),
    comment TEXT NOT NULL,
    is_verified TINYINT DEFAULT 0,
    created_at DATETIME DEFAULT NOW(),
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (artwork_id) REFERENCES artworks(id),
    FOREIGN KEY (event_id) REFERENCES events(id)
);

CREATE TABLE review_votes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    review_id INT NOT NULL,
    user_id INT NOT NULL,
    is_helpful TINYINT DEFAULT 1,
    UNIQUE (review_id, user_id),
    FOREIGN KEY (review_id) REFERENCES reviews(id),
    FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE review_replies (
    id INT AUTO_INCREMENT PRIMARY KEY,
    review_id INT UNIQUE NOT NULL,
    admin_id INT NOT NULL,
    reply TEXT NOT NULL,
    created_at DATETIME DEFAULT NOW(),
    updated_at DATETIME DEFAULT NOW() ON UPDATE NOW(),
    FOREIGN KEY (review_id) REFERENCES reviews(id),
    FOREIGN KEY (admin_id) REFERENCES users(id)
);

CREATE TABLE artwork_views (
    id INT AUTO_INCREMENT PRIMARY KEY,
    artwork_id INT NOT NULL,
    viewed_at DATETIME DEFAULT NOW(),
    FOREIGN KEY (artwork_id) REFERENCES artworks(id)
);

SET FOREIGN_KEY_CHECKS = 1;

-- DOĞRULAMA — Kurulum sonrası kontrol sorguları
-- ═══════════════════════════════════════════════════════════════════
SELECT 'users'          AS tablo, COUNT(*) AS adet FROM users
UNION ALL SELECT 'artists',        COUNT(*) FROM artists
UNION ALL SELECT 'categories',     COUNT(*) FROM categories
UNION ALL SELECT 'artworks',       COUNT(*) FROM artworks
UNION ALL SELECT 'events',         COUNT(*) FROM events
UNION ALL SELECT 'coupons',        COUNT(*) FROM coupons
UNION ALL SELECT 'reservations',   COUNT(*) FROM reservations
UNION ALL SELECT 'orders',         COUNT(*) FROM orders
UNION ALL SELECT 'reviews',        COUNT(*) FROM reviews
UNION ALL SELECT 'review_votes',   COUNT(*) FROM review_votes
UNION ALL SELECT 'support_tickets',COUNT(*) FROM support_tickets
UNION ALL SELECT 'artwork_views',  COUNT(*) FROM artwork_views;
